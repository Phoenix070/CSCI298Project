SQL to MongoDB conversion statements

The following table gives a basic overview of SQL to MongoDB query translations. This table is provided to help understand the conversion process.

Create table users(
   id varchar(30),
   name varchar(30),
   PRIMARY KEY(id)
)		 db.createCollection("users")
This will create a collection "users" with no columns.
      OR
db.users.insert( {
   id:"az2345",
   name: "Pooja Pawara"
} )
This will automatically create a collection "users" and insert one document.
Drop table users db.users.drop()
Insert into users (id, name, gender)
values("ab1234","John Doe", "Male")		 db.users.insert(
   {id: "ab1234", name: "John Doe", gender: "Male"}
)
Select * from users db.users.find()
   OR
db.users.find( {} )
Select id, name from users db.users.find( { },
   {id:1, name:1},
)
Select * from users
where age = 33		 db.users.find(
   {age: "33"}
)
Select * from users
where name LIKE "%Doe%"		 db.users.find(
   {name: /Doe/}
)
Select * users
where age > 33 AND age <= 40		 db.users.find(
   {age: {$gt: 33, $lte: 40}}
)
Select * users
order by name desc		 db.users.find().sort( {name: -1} )
This sorts the documents by name, -1: descending, 1: ascending
Select count(*) from users db.users.count()
   OR
db.users.find().count()
Select * from users
limit 1		 db.users.findOne()
   OR
db.users.find().limit(1)
Select * from users
skip 10		 db.users.find().skip(10)
Select Distinct(name) from users db.users.distinct("name")
Update users
Set status = "A"
where age > 35		 db.users.update(
   {age: { $gt: 35 }},
   {$set: {status: "A"}},
   {multi: true}
)
Update users
Set salary = salary + 1000
where status = "A"		 db.users.update(
   {status: "A" },
   {$inc: {salary: 1000}},
   {multi: true}
)
Delete from users db.users.remove( {} )
Delete from users
where status = "D"		 db.users.remove( { status: "D" } )
